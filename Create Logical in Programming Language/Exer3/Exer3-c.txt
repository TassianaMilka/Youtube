//Início
#include <stdio.h>

int main()
{
    //Declaração de variáveis
    int num=8;
    int i;
    //Imprimir a mensagem
    
    printf("Número secreto do loop\n");

 
   //Loop 
   
   for(i=0;i<2;i++){
       
     //Duas vezes imprimir o número 
     
     printf("%d\n",num);
     
   }
   
  
    return 0;
    //Finalização
}
